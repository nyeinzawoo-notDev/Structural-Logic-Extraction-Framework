# nstf_integration_engine.py - Production Ready

import pandas as pd
import json
from typing import Dict, List, Optional

class NSTFIntegrationEngine:
    def __init__(self, csv_path: str = "NSTF_Full_Framework.csv"):
        self.df = pd.read_csv(csv_path)
        self.character_map = self._create_character_map()
        self.tcode_relationships = self._initialize_tcode_relationships()
    
    def _create_character_map(self) -> Dict:
        """Create quick lookup dictionary for characters"""
        return {
            row['Character']: {
                'type': row['Type'],
                'phonetic': row['Phonetic_Code'],
                'meaning_en': row['Meaning_Essence_EN'],
                'meaning_mm': row['Meaning_Essence_MM'],
                't_code': row['T_Code'],
                'polarity_negative': row['Polarity_Negative'],
                'polarity_positive': row['Polarity_Positive'],
                'optimal_range': row['Optimal_Range'],
                'cultural_contexts': row['Cultural_Contexts'].split(',')
            }
            for _, row in self.df.iterrows()
        }
    
    def _initialize_tcode_relationships(self) -> Dict:
        """Initialize T-Code relational logic"""
        return {
            "T003": {
                "opposing": "T017",
                "complementary": ["T005", "T007"],
                "conflict_level": "HIGH",
                "balanced_expression": "Innovation with Ethical Boundaries"
            },
            "T017": {
                "opposing": "T003", 
                "complementary": ["T009", "T015"],
                "conflict_level": "HIGH",
                "balanced_expression": "Structure with Flexibility"
            },
            "T005": {
                "opposing": "T007",
                "complementary": ["T003", "T009"],
                "conflict_level": "MEDIUM",
                "balanced_expression": "Connection with Independence"
            }
        }
    
    def analyze_myanmar_text(self, text: str, context: str = "universal") -> Dict:
        """Analyze Myanmar text using NSTF framework"""
        
        analysis_result = {
            "text": text,
            "character_analysis": [],
            "tcode_distribution": {},
            "ethical_balance": {},
            "cultural_appropriateness": {},
            "overall_verdict": "NEUTRAL"
        }
        
        # Analyze each character
        for char in text:
            if char in self.character_map:
                char_data = self.character_map[char]
                
                # Character-level analysis
                char_analysis = {
                    "character": char,
                    "phonetic": char_data['phonetic'],
                    "meaning_en": char_data['meaning_en'],
                    "meaning_mm": char_data['meaning_mm'],
                    "t_code": char_data['t_code'],
                    "cultural_fit": self._check_cultural_fit(char_data, context)
                }
                
                analysis_result["character_analysis"].append(char_analysis)
                
                # Update T-Code distribution
                if char_data['t_code'] != '-':
                    t_code = char_data['t_code']
                    analysis_result["tcode_distribution"][t_code] = \
                        analysis_result["tcode_distribution"].get(t_code, 0) + 1
        
        # Calculate ethical balance
        analysis_result["ethical_balance"] = self._calculate_ethical_balance(
            analysis_result["tcode_distribution"]
        )
        
        # Determine overall verdict
        analysis_result["overall_verdict"] = self._determine_verdict(
            analysis_result["ethical_balance"]
        )
        
        return analysis_result
    
    def _check_cultural_fit(self, char_data: Dict, context: str) -> str:
        """Check if character essence fits cultural context"""
        if context in char_data['cultural_contexts'] or 'universal' in char_data['cultural_contexts']:
            return "HIGH_COMPATIBILITY"
        elif any(ctx in char_data['cultural_contexts'] for ctx in [context, 'universal']):
            return "MEDIUM_COMPATIBILITY"
        else:
            return "LOW_COMPATIBILITY"
    
    def _calculate_ethical_balance(self, tcode_distribution: Dict) -> Dict:
        """Calculate ethical balance based on T-Code distribution"""
        
        total_chars = sum(tcode_distribution.values())
        if total_chars == 0:
            return {"balance_score": 0.5, "primary_conflict": "NONE"}
        
        # Calculate T003 vs T017 balance (Expansion vs Containment)
        t003_count = tcode_distribution.get('T003', 0)
        t017_count = tcode_distribution.get('T017', 0)
        
        expansion_ratio = t003_count / total_chars if total_chars > 0 else 0
        containment_ratio = t017_count / total_chars if total_chars > 0 else 0
        
        balance_score = 1 - abs(expansion_ratio - containment_ratio)
        
        # Identify primary conflicts
        conflicts = []
        if expansion_ratio > 0.7 and containment_ratio < 0.3:
            conflicts.append("T003_T017_CRITICAL")
        elif expansion_ratio > 0.6:
            conflicts.append("T003_DOMINANCE")
        elif containment_ratio > 0.6:
            conflicts.append("T017_DOMINANCE")
        
        return {
            "balance_score": balance_score,
            "expansion_ratio": expansion_ratio,
            "containment_ratio": containment_ratio,
            "conflicts": conflicts,
            "recommendation": self._generate_balance_recommendation(expansion_ratio, containment_ratio)
        }
    
    def _generate_balance_recommendation(self, expansion: float, containment: float) -> str:
        """Generate balance recommendations based on T-Code ratios"""
        
        if expansion > 0.7 and containment < 0.3:
            return "Critical imbalance: Introduce T017 ethical boundaries to contain aggressive expansion"
        elif expansion > 0.6:
            return "Moderate expansion dominance: Balance with T005 collaborative approaches"
        elif containment > 0.6:
            return "High containment: Consider T003 innovative energy to prevent stagnation"
        elif abs(expansion - containment) < 0.2:
            return "Well balanced: Maintain current harmony between growth and ethics"
        else:
            return "Moderately balanced: Minor adjustments may enhance ethical alignment"
    
    def _determine_verdict(self, ethical_balance: Dict) -> str:
        """Determine overall ethical verdict"""
        
        balance_score = ethical_balance.get("balance_score", 0.5)
        conflicts = ethical_balance.get("conflicts", [])
        
        if "T003_T017_CRITICAL" in conflicts:
            return "ETHICAL_CONCERN"
        elif balance_score > 0.7 and not conflicts:
            return "ETHICALLY_SOUND"
        elif balance_score > 0.5:
            return "MODERATELY_BALANCED"
        else:
            return "NEEDS_ADJUSTMENT"
